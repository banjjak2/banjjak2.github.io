class Calculator<T> {
    fun add(a: T, b: T, op: (T, T) -> T): T {
        return op(a, b)
    }
}

fun main() {
    val calculator = Calculator<Int>()
    val data = calculator.add(10, 20) { a, b ->
        a + b
    }
    println(data)

    val calculator2 = Calculator<String>()
    val data2 = calculator2.add("a", "b") { a, b ->
        a + b
    }
    println(data2)
}
